import torch.nn as nn

class VAE(nn.Module):
    def __init__(self, input_dim, latent_dim):
        super(VAE, self).__init__()
        self.encoder = nn.Sequential(
            nn.Linear(input_dim, 128),
            nn.ReLU(),
            nn.Linear(128, 64),
            nn.ReLU()
        )
        self.mu = nn.Linear(64, latent_dim)
        self.log_var = nn.Linear(64, latent_dim)

        self.decoder = nn.Sequential(
            nn.Linear(latent_dim, 64),
            nn.ReLU(),
            nn.Linear(64, 128),
            nn.ReLU(),
            nn.Linear(128, input_dim),
            nn.Sigmoid()
        )

    def encode(self, x):
        h = self.encoder(x)
        return self.mu(h), self.log_var(h)

    def reparameterize(self, mu, log_var):
        std = torch.exp(0.5 * log_var)
        eps = torch.randn_like(std)
        return mu + eps * std

    def decode(self, z):
        return self.decoder(z)

    def forward(self, x):
        mu, log_var = self.encode(x)
        z = self.reparameterize(mu, log_var)
        return self.decode(z), mu, log_var

vae = VAE(input_dim=features_tensor.shape[1], latent_dim=5)
optimizer = torch.optim.Adam(vae.parameters(), lr=0.001)
loss_function = nn.MSELoss()

epochs = 50
for epoch in range(epochs):
    vae.train()
    train_loss = 0
    for batch in train_loader:
        batch = batch[0]
        optimizer.zero_grad()
        reconstructed, mu, log_var = vae(batch)

        # Loss calculation
        reconstruction_loss = loss_function(reconstructed, batch)
        kl_divergence = -0.5 * torch.sum(1 + log_var - mu.pow(2) - log_var.exp()) / batch.size(0)
        loss = reconstruction_loss + kl_divergence

        loss.backward()
        optimizer.step()
        train_loss += loss.item()

    print(f'Epoch {epoch+1}, Loss: {train_loss / len(train_loader)}')
